```typescript
// src/types/pdf.ts — target shape
export type ExtractResult = {
  text: string;
  pageCount: number;
};

// src/lib/pdf/validatePdfFile.ts — target API
export type PdfValidationError = "type" | "size";

export function validatePdfFile(file: File):
  | { ok: true }
  | { ok: false; error: PdfValidationError };
```

Constants (concrete values — use exactly):
- `MAX_PDF_BYTES = 10 * 1024 * 1024` (10 MiB)
- Accept if `file.type === "application/pdf"` OR `file.name.toLowerCase().endsWith(".pdf")` (both checks for resilience)
- Reject with `error: "size"` when `file.size > MAX_PDF_BYTES`
- Reject with `error: "type"` when neither PDF condition passes
</interfaces>
